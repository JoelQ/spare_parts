Light from above for type

★

Code

HTML

<div class="style-light-from-above">
  <p>&#9733;</p>
</div>

CSS

@mixin text-gradient ($top, $bottom) {
  color: $bottom;
  background: -webkit-linear-gradient($top, $bottom);
  -webkit-background-clip: text;
  -webkit-text-fill-color: transparent;
}

@mixin blend-layer($color, $edge-sharpness, $transparentize) {
  text-shadow: 0 1px $edge-sharpness transparentize($color, $transparentize);
}

.style-light-from-above {
  @include blend-layer(red, 2px, 0.5);
  @include text-gradient(white, darken(red, 10));

  p {
    font-size: 60px;
    margin: 0;
    line-height: 60px;
  }
}

Tooltip text

HTML

<a href="#" class="item-one">
  Hover for Tooltip

</a>

CSS

@mixin ui-nub($size: 8px, $position: top-right,  $bg-color: #fff, $border-color: rgb(100,100,100)) {

  &:before,
  &:after {
    display: block;
    pointer-events: none;
    position: absolute;
    width: 0;
    height: 0;
    content: '';
  }

  &:before {
    border: ($size + 1) solid transparent;
  }

  &:after {
    border: $size solid transparent;
  }

  @if ($position == bottom-left) or ($position == bottom-right) or ($position == bottom-center) {
    &:before {
      border-top-color: $border-color;
      top: auto;
      bottom: -(($size * 2) + 2);
      left: auto;
      right: auto;
    }

    &:after {
      border-top-color: $bg-color;
      top: auto;
      bottom: -(($size * 2) - 0);
      left: auto;
      right: auto;
    }

    @if $position == bottom-center {
      &:before {
        left: 50%;
        margin-left: -($size + 1);
      }

      &:after {
        left: 50%;
        margin-left: -$size;
      }
    }
  }
}

@mixin ui-tooltip($width: 200px, $position: bottom, $bg-color: rgb(80,80,80), $tooltip-selector: div) {
  position: relative;

  &:hover {
    #{$tooltip-selector} {
      opacity: 1;
      visibility: visible;
    }
  }

  #{$tooltip-selector} {
    background-color: $bg-color;
    box-sizing: border-box;
    color: white;
    font-weight: normal;
    letter-spacing: 0px;
    opacity: 0;
    padding: 5px 8px 6px;
    border-radius: 3px;
    position: absolute;
    text-align: center;
    text-overflow: ellipsis;
    text-shadow: 0 1px 0 darken($bg-color, 4%);
    text-transform: none;
    transition: all 0.15s linear;
    white-space: nowrap;
    visibility: hidden;
    width: $width;
    z-index: 10;

    @if ($position == bottom) or ($position == top) {
      #{$position}: -50px;
      left: 50%;
      margin-left: -($width/2);
    }

    @else if ($position == left) or ($position == right) {
      #{$position}: (-$width) - 15px;
      margin-bottom: -17px;
      bottom: 50%;
    }

    @include ui-nub(8px, bottom-center, $bg-color, none);



    &:hover {
      opacity: 0;
      visibility: hidden;
    }
  }
}

a.item-one {

  background-color: gray;
  padding: 5px 8px;;
}
