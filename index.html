@mixin text-stroke ($color, $width) {
  -webkit-text-stroke: $width $color;

}
@mixin text-gloss ($color-top, $color-top-mid, $color-bottom-mid, $color-bottom) {
  color: $color-bottom-mid;
  background: -webkit-gradient(linear, left top, left bottom, 
    color-stop(0%,$color-top), 
    color-stop(50%,$color-top-mid), 
    color-stop(51%,$color-bottom-mid), 
    color-stop(100%,$color-bottom));
  -webkit-background-clip: text;
  -webkit-text-fill-color: transparent;
}
.style-gloss { // four steps in the gradient for glossy metalic

@mixin ui-nub($size: 8px, $position: top-right,  $bg-color: #fff, $border-color: rgb(100,100,100)) {

  &:before,
  &:after {
    display: block;
    pointer-events: none;
    position: absolute;
    width: 0;
    height: 0;
    content: '';
  }

  &:before {
    border: ($size + 1) solid transparent;
  }

  &:after {
    border: $size solid transparent;
  }

  @if ($position == bottom-left) or ($position == bottom-right) or ($position == bottom-center) {
    &:before {
      border-top-color: $border-color;
      top: auto;
      bottom: -(($size * 2) + 2);
      left: auto;
      right: auto;
    }

    &:after {
      border-top-color: $bg-color;
      top: auto;
      bottom: -(($size * 2) - 0);
      left: auto;
      right: auto;
    }

    @if $position == bottom-center {
      &:before {
        left: 50%;
        margin-left: -($size + 1);
      }

      &:after {
        left: 50%;
        margin-left: -$size;
      }
    }
  }
}

@mixin ui-tooltip($width: 200px, $position: bottom, $bg-color: rgb(80,80,80), $tooltip-selector: div) {
  position: relative;

  &:hover {
    #{$tooltip-selector} {
      opacity: 1;
      visibility: visible;
    }
  }

  #{$tooltip-selector} {
    background-color: $bg-color;
    box-sizing: border-box;
    color: white;
    font-weight: normal;
    letter-spacing: 0px;
    opacity: 0;
    padding: 5px 8px 6px;
    border-radius: 3px;
    position: absolute;
    text-align: center;
    text-overflow: ellipsis;
    text-shadow: 0 1px 0 darken($bg-color, 4%);
    text-transform: none;
    transition: all 0.15s linear;
    white-space: nowrap;
    visibility: hidden;
    width: $width;
    z-index: 10;

    @if ($position == bottom) or ($position == top) {
      #{$position}: -50px;
      left: 50%;
      margin-left: -($width/2);
    }

    @else if ($position == left) or ($position == right) {
      #{$position}: (-$width) - 15px;
      margin-bottom: -17px;
      bottom: 50%;
    }

    @include ui-nub(8px, bottom-center, $bg-color, none);



    &:hover {
      opacity: 0;
      visibility: hidden;
    }
  }
}
